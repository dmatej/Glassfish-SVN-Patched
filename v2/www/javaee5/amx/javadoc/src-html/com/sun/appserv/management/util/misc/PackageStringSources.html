023    package com.sun.appserv.management.util.misc;
024    
025    import java.lang.reflect.Constructor;
026    import java.util.ResourceBundle;
027    import java.util.Locale;
028    import java.util.HashMap;
029    import java.util.Map;
030    
031    
032    import com.sun.appserv.management.util.misc.ClassUtil;
033    
034    /**
035            A StringSource for a package which first looks for a class in the the package
036            named "PackageStringSource", and then, if not found, attempts to locate
037            a ResourceBundle with the name "Strings".
038            <p>
039            Also maintains a cache of such sources so that subsequent uses reuse prior ones.
040     */
041    public class PackageStringSources
042    {
043            static final Map<Class,StringSource> mSources     = new HashMap<Class,StringSource>();
044            
045            /**
046                    Get a string source for the specified class, using the specified
047                    StringSource as its delegate.
048             */
049                    public static StringSource
050            get( final Class theClass, final StringSource delegate )
051            {
052                    StringSource    source  = mSources.get( theClass );
053                    if ( source == null )
054                    {
055                            source  = init( theClass, delegate );
056                            mSources.put( theClass, source );
057                    }
058    
059                    return( source );
060            }
061            
062            private PackageStringSources()  {}
063            
064                    private static StringSource
065            init( final Class theClass,  final StringSource delegate )
066            {
067                    StringSource    source  = null;
068                    final String    packageName     = theClass.getPackage().getName();
069                    
070                    try
071                    {
072                            final String    classname       = packageName + ".PackageStrings";
073                            
074                            final Class     packageStringSourceClass        = ClassUtil.getClassFromName( classname );
075                            
076                            final Constructor       c       = packageStringSourceClass.getConstructor( new Class[] { StringSource.class } );
077                            
078                            source  = (StringSource)c.newInstance( new Object[] { delegate } );
079                    }
080                    catch( Exception e )
081                    {
082                            // no such class exists
083                            e.printStackTrace();
084                    }
085                    
086                    if ( source == null )
087                    {
088                            final ResourceBundle bundle     = ResourceBundle.getBundle( packageName + ".Strings",
089                                                                                    Locale.getDefault(), theClass.getClassLoader());
090                            
091                            source  = new ResourceBundleStringSource( bundle, delegate );
092                    }
093                    
094                    return( source );
095            }
096    };
