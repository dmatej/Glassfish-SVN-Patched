025    package com.sun.appserv.management.config;
026    
027    import java.util.Map;
028    import java.util.List;
029    
030    import com.sun.appserv.management.base.XTypes;
031    import com.sun.appserv.management.base.Container;
032    
033    /**
034     * Config AMX MBean for Web Service Endpoint's config
035     * 
036     * @since AppServer 9.0
037     */
038    public interface WebServiceEndpointConfig extends NamedConfigElement, Container
039    {
040        public static final String  J2EE_TYPE       = XTypes.WEB_SERVICE_ENDPOINT_CONFIG;
041    
042        /**
043        * Get the web service name.
044        */
045        public String getName();
046    
047        /**
048        * visibility of this endoint as a service in JBI
049        */
050        public boolean getJBIEnabled();
051    
052        /**
053        * visibility of this endoint as a service in JBI
054        */
055        public void setJBIEnabled( boolean enabled );
056    
057        /**
058        * Get the web service' monitoring level can be OFF, LOW or HIGH.
059        @see ModuleMonitoringLevelValues
060        */
061        public String getMonitoringLevel();
062    
063        /**
064        * Sets the web service' monitoring level can be OFF, LOW or HIGH.
065        @see ModuleMonitoringLevelValues
066        */
067        public void setMonitoringLevel( String level );
068    
069        /**
070        * Gets the web service' max history size. Number of statistics stored in
071        * memory.
072        */
073        public String getMaxHistorySize();
074    
075        /**
076        * Sets the web service' max history size. Number of statistics stored in
077        * memory.
078        *
079        * @param maxSize max size of stored statistics in memory
080        */
081        public void setMaxHistorySize(String maxSize);
082        
083        /**
084            @since Appserver 9.0
085         */
086        public Map<String,RegistryLocationConfig> getRegistryLocationConfigMap();
087        
088        
089        /**
090            @since Appserver 9.0
091         */
092        public RegistryLocationConfig createRegistryLocationConfig( String jndiName );
093        
094        /**
095            @since Appserver 9.0
096         */
097        public void removeRegistryLocationConfig( String jndiName );
098        
099    
100        /**
101        * Get the transformation rules (not in any specific order) defined for this
102        * end point.
103        *
104        * @return Map of transformation rules added to web service.
105        */
106        public Map<String,TransformationRuleConfig> getTransformationRuleConfigMap();
107        /**
108        * Get the transformation rules (in the same order as in domain.xml) 
109        * defined for this end point.
110        *
111        * @return Map of transformation rules added to web service.
112        */
113        public List<TransformationRuleConfig> getTransformationRuleConfigList();
114    
115    
116        /**
117        * Removes the specified transformation rule for this end point.
118        *
119        * @param name Name of the transformation rule
120        */
121        public void removeTransformationRuleConfig(String name);
122    
123        /**
124        * Creates a transformation rule for this web service.
125        *
126        * @param name  this transformation rule's name
127        * @param ruleFileLocation File path location of this transformation rule
128        * @param applyTo Operation(s) for which this
129        * transformation rule is applied. Possible values are "request", "response"
130        * or "both"
131        * @param enabled If true transformation rule is applied
132        *
133        * @return TransformationRule new transformation rule with the
134        * specified parameters
135        */
136        public TransformationRuleConfig createTransformationRuleConfig(
137             String name,  String ruleFileLocation,boolean enabled, String applyTo,
138             Map<String,String> reserved);
139    }
