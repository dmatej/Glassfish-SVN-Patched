023    package com.sun.appserv.management.deploy;
024    
025    import java.util.Map;
026    import java.util.List;
027    import java.util.Iterator;
028    import java.io.Serializable;
029    
030    import com.sun.appserv.management.base.OperationStatus;
031    
032    /**
033            Final status of a deployment.
034            
035            See {@link com.sun.appserv.management.deploy.DeploymentMgr}.getFinalDeploymentStatus()
036     */
037    public interface DeploymentStatus extends OperationStatus
038    {
039            /**
040                    Value of the MapCapable.MAP_CAPABLE_CLASS_NAME_KEY when turned into a Map.
041             */
042            public final static String      DEPLOYMENT_STATUS_CLASS_NAME    = 
043                            "com.sun.appserv.management.deploy.DeploymentStatus";
044            
045            /**
046                    Status code indicating failure due to improper initialization.
047             */
048            public final int        STATUS_CODE_NOT_INITIALIZED = 0;
049            
050            
051            public static final String      STAGE_STATUS_KEY                = STATUS_CODE_KEY;
052            
053            public static final String      STAGE_STATUS_MESSAGE_KEY = "StageStatusMessage";
054            
055            public static final String      SUB_STAGES_KEY                  = "SubStages";
056            
057            public static final String      STAGE_THROWABLE_KEY             = THROWABLE_KEY;
058            
059            public static final String      STAGE_DESCRIPTION_KEY    = "StageDescription";
060            
061            public static final String      PARENT_KEY               = "Parent";
062    
063            public static final String      ADDITIONAL_STATUS_KEY               = "AdditionalStatus";
064            
065            /**
066              * Key within the Map of AdditionalStatus. 
067              * The unique module id of the deployed component.
068              */     
069            public static final String      MODULE_ID_KEY               = "ModuleID";
070    
071            /**
072                Get the sub stages for this deployment status, each of
073                which has been converted into a Map<String,Serializable>.
074                This method is included for backward compatibility; please
075                use {@link #getSubStagesList} instead.
076                @return an Iterator for the sub stages
077                @deprecated
078             */
079            public Iterator<Map<String,Serializable>> getSubStages();
080            
081            /**
082                Get the sub stages for this deployment status.
083                The list may not be modified.
084                @return List&lt;DeploymentStatus>
085             */
086            public List<DeploymentStatus> getSubStagesList();
087            
088            /** 
089                    Legal status codes include:
090                    <ul>
091                    <li>OperationStatus#STATUS_CODE_SUCCESS</li>
092                    <li>OperationStatus#STATUS_CODE_FAILURE</li>
093                    <li>OperationStatus#STATUS_CODE_WARNING</li>
094                    <li>#STATUS_CODE_NOT_INITIALIZED</li>
095                    </ul>
096                    @return the status for this stage (ignoring sub stages status)
097             */
098            public int getStageStatus();
099            
100            /**
101             * @return a meaningful i18ned reason for failure or warning
102             */
103            public String getStageStatusMessage();
104            
105            /**
106             * @return the exception if an exception was thrown during 
107             *  the execution of the stage
108             */
109            public Throwable getStageThrowable();
110    
111            /**
112             * @return a meaningful i18ned stage description
113             */
114            public String getStageDescription();
115    
116            /**
117             * @return the map storing additional properties for this status
118             */
119            public Map<String,Serializable> getAdditionalStatus();
120    
121        /**
122         * @return the parent status for this status if any
123         */
124        public DeploymentStatus getParent();
125    
126        /**
127         * set the parent status for this status if any
128         * @param parent DeploymentStatus
129         */
130        public void setParent( DeploymentStatus parent );
131    }
