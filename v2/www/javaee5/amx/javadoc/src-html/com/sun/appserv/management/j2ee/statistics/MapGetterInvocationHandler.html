023    package com.sun.appserv.management.j2ee.statistics;
024    
025    import java.util.Map;
026    import java.io.Serializable;
027    
028    import com.sun.appserv.management.util.misc.MapUtil;
029    import com.sun.appserv.management.util.misc.ObjectUtil;
030    import com.sun.appserv.management.util.misc.TypeCast;
031    
032    /**
033            Implements getXXX() based on a Map whose keys are the XXX part of the getXXX() method name.
034            Serializable so that it may be used to return a result remotely.
035            <br><b>Internal use only</b>
036     */
037    public class MapGetterInvocationHandler<T>
038            extends GetterInvocationHandler<T>
039            implements Serializable 
040    {
041            static final long serialVersionUID = -8751876448821319456L;
042    
043            private final Map<String,T>       mMap;
044            
045            /**
046                    Create a new instance using the Map, which is <b>not</b> copied.
047             */
048                    public
049            MapGetterInvocationHandler( final Map<String,T> map )
050            {
051                    mMap    = map;
052            }
053            
054                    protected Map<String,T>
055            getMap()
056            {
057                    return( mMap );
058            }
059            
060                    protected T
061            getValue( final String name )
062            {
063                    return( mMap.get( name ) );
064            }
065    
066                    protected boolean
067            containsValue( String name )
068            {
069                    return( mMap.containsKey( name ) );
070            }
071            
072                public int
073            hashCode()
074            {
075                return ObjectUtil.hashCode( mMap );
076            }
077            
078                    public boolean
079            equals( final Object rhsIn )
080            {
081                    boolean equals  = false;
082                    
083                    if ( rhsIn instanceof MapGetterInvocationHandler )
084                    {
085                        final MapGetterInvocationHandler<?> rhs = 
086                            MapGetterInvocationHandler.class.cast( rhsIn );
087                            
088                            equals  = MapUtil.mapsEqual( getMap(), rhs.getMap() );
089                    }
090                     
091                    return( equals );
092            }
093            
094            
095                    public String
096            toString( )
097            {
098                    return( MapUtil.toString( mMap ) );
099            }
100    }
