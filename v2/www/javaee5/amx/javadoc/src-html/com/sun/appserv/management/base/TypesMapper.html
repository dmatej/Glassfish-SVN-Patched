023    package com.sun.appserv.management.base;
024    
025    import java.util.Map;
026    import java.util.Set;
027    import java.util.HashSet;
028    import java.util.Collections;
029    import java.util.HashMap;
030    import java.lang.reflect.Field;
031    
032    
033    /**
034            Map all types from XTypes to their respective MBean interfaces.
035            
036            @see AllTypesMapper
037            @see com.sun.appserv.management.j2ee.J2EETypesMapper
038     */
039    public class TypesMapper
040    {
041            private final Map<String,Class>   mMap;
042            
043                    public
044            TypesMapper( final Class[] interfaces )
045            {
046                    mMap    = init( interfaces );
047            }
048            
049            /**
050                    Map all types to interfaces.
051             */
052                    private Map<String,Class>
053            init( final Class[] interfaces )
054            {
055                    final Map<String,Class>   m       = new HashMap<String,Class>();
056    
057                    for( int i = 0; i < interfaces.length; ++i )
058                    {
059                            final Class     theInterface    = interfaces[ i ];
060                            
061                            try
062                            {
063                                    final Field     field   = theInterface.getField( "J2EE_TYPE" );
064                                    final String value      = (String)field.get( theInterface );
065                                    if ( m.containsKey( value ) )
066                                    {
067                                            final String    msg     =
068                                                    "TypesMapper: key already present: " +
069                                                    value + " for " + theInterface.getName();
070                                            
071                                            assert(  false ): msg;
072                                            throw new RuntimeException( msg );
073                                    }
074                                    m.put( value, theInterface );
075                            }
076                            catch( Exception e )
077                            {
078                                    e.printStackTrace();
079                                    assert( false );
080                                    throw new IllegalArgumentException( theInterface.getName() );
081                            }
082                    }
083                    
084                    return( m );
085            }
086            
087            /**
088                    Return the Class associated with a given type.
089             */
090                    public Class
091            getInterfaceForType( final String type )
092            {
093                    final Class theClass    = mMap.get( type );
094                    
095                    return( theClass );
096            }
097            
098                public Set<String>
099            getJ2EETypes()
100            {
101                return Collections.unmodifiableSet( mMap.keySet() );
102            }
103            
104                public Set<Class>
105            getClasses()
106            {
107                final Set<Class>    classes = new HashSet<Class>();
108                classes.addAll( mMap.values() );
109                return Collections.unmodifiableSet( classes );
110            }
111    }
