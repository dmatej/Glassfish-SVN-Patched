023    package com.sun.appserv.management.deploy;
024    
025    import java.util.Locale;
026    import java.util.Map;
027    import java.io.Serializable;
028    
029    
030    import com.sun.appserv.management.deploy.DeploymentProgress;
031    import com.sun.appserv.management.base.MapCapableBase;
032    
033    /**
034            Use DeploymentSupport to create a new instance of this class.
035     */
036    public final class DeploymentProgressImpl
037            extends MapCapableBase
038            implements DeploymentProgress
039    {       
040            /**
041                    Create a new instance based on another.
042             */
043                    public
044            DeploymentProgressImpl( final DeploymentProgress src )
045            {
046                    this( src.asMap() );
047            }
048            
049                    protected boolean
050            validate()
051            {
052                    final byte      progressPercent = getProgressPercent();
053                    return( progressPercent >= 0 && progressPercent <= 100 );
054            }
055            
056            /**
057                    Create a new instance with explicit params
058                    
059                    @param progressPercent
060                    @param description
061                    @param other other values, see this( Map m )
062             */
063                    public <T extends Serializable>
064            DeploymentProgressImpl(
065                    final byte              progressPercent,
066                    final String    description,
067                    final Map<String,T> other )
068            {
069                    this( other, false );
070                    
071                    putField( PROGRESS_PERCENT_KEY, new Byte( (byte)progressPercent ) );
072                    putField( DESCRIPTION_KEY, description );
073                    
074                    validateThrow();
075            }
076            
077            /**
078                    @param m        a Map representing a DeploymentProgress
079                    @param validate true if should validate
080             */
081                    private <T extends Serializable>
082            DeploymentProgressImpl( final Map<String,T> m, final boolean validate )
083            {
084                    super( m, DEPLOYMENT_PROGRESS_CLASS_NAME);
085    
086                    if ( validate )
087                    {
088                            validateThrow();
089                    }
090            }
091            
092            /**
093                    @param data     a Map representing a DeploymentProgress
094             */
095            /**
096                    Create a new instance.  The Map must contain the following
097                    keyed values:
098                    
099                    <ul>
100                    <li>{@link com.sun.appserv.management.base.MapCapable}.MAP_CAPABLE_TYPE_KEY with
101                            value DEPLOYMENT_PROGRESS_CLASS_NAME</li>
102                    <li>PROGRESS_PERCENT_KEY</li>
103                    <li>DESCRIPTION_KEY</li>
104                    </ul>
105                    <p>
106                    The map may contain also contain localized descriptions.
107                    See {@link DeploymentProgress}.LOCALIZED_DESCRIPTION_KEY_BASE.
108                    
109                    @param m        a Map representing a DeploymentProgress
110             */
111                    public <T extends Serializable>
112            DeploymentProgressImpl( final Map<String,T> m )
113            {
114                    super( m, DEPLOYMENT_PROGRESS_CLASS_NAME );
115                    checkValidType( m, DEPLOYMENT_PROGRESS_CLASS_NAME );
116                    
117                    validateThrow();
118            }
119            
120                    public String
121            getMapClassName()
122            {
123                    return( DEPLOYMENT_PROGRESS_CLASS_NAME );
124            }
125            
126                    public static String
127            getLocalizedDescriptionKey( final Locale locale )
128            {
129                    return( LOCALIZED_DESCRIPTION_KEY_BASE + "_" + locale.toString() );
130            }
131        
132            public byte
133        getProgressPercent()
134        {
135            return( getByte( PROGRESS_PERCENT_KEY ).byteValue() );
136        }
137        
138            public String
139        getDescription()
140        {
141            return( getString( DESCRIPTION_KEY ) );
142        }
143        
144                    public String
145        getLocalizedDescription( final Locale locale)
146        {
147            return( getString( getLocalizedDescriptionKey( locale ) ) );
148        }
149    }
