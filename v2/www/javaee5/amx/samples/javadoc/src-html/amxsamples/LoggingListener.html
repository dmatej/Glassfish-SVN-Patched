029    /**
030            A JMX NotificationListener for deployment.
031     */
032    public final class LoggingListener
033                    implements NotificationListener
034    {
035        private final Set<String>         mTypes;
036        private final List<Notification>  mHistory;
037        
038                    public
039            LoggingListener()
040            {
041                mTypes  = GSetUtil.newStringSet( Logging.LOG_RECORD_SEVERE_NOTIFICATION_TYPE,
042                                Logging.LOG_RECORD_WARNING_NOTIFICATION_TYPE,
043                                Logging.LOG_RECORD_INFO_NOTIFICATION_TYPE );
044                
045                mHistory    = new CircularList<Notification>( Notification.class, 100 );
046            }
047            
048                    public synchronized void
049            handleNotification(
050                    final Notification      notif, 
051                    final Object            handback) 
052            {
053                    try
054                    {
055                            realHandleNotification( notif, handback );
056                    }
057                    catch( final Exception e )
058                    {
059                            e.printStackTrace();
060                    }
061            }
062            
063                public void
064            setNotificationTypes( final Set<String> types )
065            {
066                mTypes.clear();
067                mTypes.addAll( types );
068            }
069            
070                public Set<String>
071            getNotificationTypes( )
072            {
073                return Collections.unmodifiableSet( mTypes );
074            }
075            
076            
077            /**
078                    Note that Notifications are not guaranteed to be delivered in order.
079                    Thus, it is theoretically possible for a DEPLOYMENT_COMPLETED_NOTIFICATION_TYPE
080                    to be received before a DEPLOYMENT_STARTED_NOTIFICATION_TYPE.
081             */
082                    public void
083            realHandleNotification(
084                    final Notification      notif, 
085                    final Object            handback) 
086            {
087                    final String    type    = notif.getType();
088                    final Map<String,Object> m        = Util.getAMXNotificationData( notif );
089                    
090                    if ( type.startsWith( Logging.LOG_RECORD_NOTIFICATION_PREFIX ) &&
091                        mTypes.contains( type ) )
092                    {
093                        SampleUtil.println( NotificationStringifier.toString( notif ) + 
094                            StringUtil.NEWLINE() + "USER DATA: " + MapUtil.toString( m ) );
095                            
096                        mHistory.add( notif );
097                    }
098            }
099            
100                public synchronized List<Notification>
101            getHistory()
102            {
103                final List<Notification>    list    = new ArrayList<Notification>();
104                
105                list.addAll( mHistory );
106                return list;
107            }
108    }
